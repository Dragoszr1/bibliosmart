CREATE DATABASE IF NOT EXISTS biblioteca;
USE biblioteca;

SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS carti;
CREATE TABLE carti (
	carte_id INTEGER NOT NULL AUTO_INCREMENT, 
	titlu VARCHAR(50) NOT NULL, 
	autor VARCHAR(50) NOT NULL, 
	`ISBN` VARCHAR(13) NOT NULL, 
	stoc_total INTEGER, 
	stoc_disponibil INTEGER, 
	imprumutat BOOL, 
	gen VARCHAR(255) NOT NULL, 
	pozitie VARCHAR(100), 
	cod VARCHAR(50), 
	PRIMARY KEY (carte_id), 
	UNIQUE (`ISBN`)
);

DROP TABLE IF EXISTS users;
CREATE TABLE users (
	user_id INTEGER NOT NULL AUTO_INCREMENT, 
	username VARCHAR(50) NOT NULL, 
	email VARCHAR(50) NOT NULL, 
	hashed_password VARCHAR(255) NOT NULL, 
	rol ENUM('user','bibliotecar') NOT NULL, 
	telefon VARCHAR(50), 
	description VARCHAR(255), 
	club BOOL NOT NULL, 
	PRIMARY KEY (user_id), 
	UNIQUE (username), 
	UNIQUE (email), 
	UNIQUE (telefon)
);

DROP TABLE IF EXISTS carti_citite;
CREATE TABLE carti_citite (
	id INTEGER NOT NULL AUTO_INCREMENT, 
	user_id INTEGER NOT NULL, 
	carte_id INTEGER NOT NULL, 
	PRIMARY KEY (id)
);

DROP TABLE IF EXISTS recenzii;
CREATE TABLE recenzii (
	id INTEGER NOT NULL AUTO_INCREMENT, 
	user_id INTEGER NOT NULL, 
	carte_id INTEGER NOT NULL, 
	nota INTEGER NOT NULL, 
	comentariu VARCHAR(255) NOT NULL, 
	PRIMARY KEY (id)
);

DROP TABLE IF EXISTS anunturi;
CREATE TABLE anunturi (
	anunt_id INTEGER NOT NULL AUTO_INCREMENT, 
	titlu VARCHAR(255) NOT NULL, 
	anunt TEXT, 
	data_publicare DATETIME NOT NULL, 
	aprecieri INTEGER, 
	PRIMARY KEY (anunt_id)
);

DROP TABLE IF EXISTS anunturi_aprecieri;
CREATE TABLE anunturi_aprecieri (
	id INTEGER NOT NULL AUTO_INCREMENT, 
	anunt_id INTEGER NOT NULL, 
	user_id INTEGER NOT NULL, 
	PRIMARY KEY (id)
);

DROP TABLE IF EXISTS cereri_carti;
CREATE TABLE cereri_carti (
	cerere_id INTEGER NOT NULL AUTO_INCREMENT, 
	user_id INTEGER NOT NULL, 
	carte_id INTEGER NOT NULL, 
	status ENUM('pending','approved','rejected') NOT NULL, 
	created_at DATETIME NOT NULL, 
	updated_at DATETIME NOT NULL, 
	PRIMARY KEY (cerere_id)
);

DROP TABLE IF EXISTS imprumuturi_active;
CREATE TABLE imprumuturi_active (
	imprumut_id INTEGER NOT NULL AUTO_INCREMENT, 
	user_id INTEGER NOT NULL, 
	carte_id INTEGER NOT NULL, 
	data_imprumut DATETIME NOT NULL, 
	data_scadenta DATETIME NOT NULL, 
	PRIMARY KEY (imprumut_id), 
	FOREIGN KEY(user_id) REFERENCES users (user_id) ON DELETE CASCADE, 
	FOREIGN KEY(carte_id) REFERENCES carti (carte_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS coduri_verificare;
CREATE TABLE coduri_verificare (
	id INTEGER NOT NULL AUTO_INCREMENT, 
	user_id INTEGER NOT NULL, 
	code VARCHAR(6) NOT NULL, 
	temp_token VARCHAR(255) NOT NULL, 
	expires_at DATETIME NOT NULL, 
	PRIMARY KEY (id),
	FOREIGN KEY(user_id) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS club_invites;
CREATE TABLE club_invites (
	token VARCHAR(255) NOT NULL, 
	created_by INTEGER NOT NULL, 
	expires_at DATETIME NOT NULL, 
	PRIMARY KEY (token),
	FOREIGN KEY(created_by) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS activitati_club;
CREATE TABLE activitati_club (
	activitate_id INTEGER NOT NULL AUTO_INCREMENT, 
	titlu VARCHAR(255) NOT NULL, 
	continut TEXT, 
	tip VARCHAR(50) DEFAULT 'activitate', 
	saptamana VARCHAR(50) DEFAULT 'curenta', 
	creat_de INTEGER NOT NULL, 
	creat_la DATETIME DEFAULT CURRENT_TIMESTAMP, 
	actualizat_la DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, 
	nume_imagine VARCHAR(255), 
	PRIMARY KEY (activitate_id),
	FOREIGN KEY(creat_de) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS comentarii_activitati;
CREATE TABLE comentarii_activitati (
	comentariu_id INTEGER NOT NULL AUTO_INCREMENT, 
	activitate_id INTEGER NOT NULL, 
	user_id INTEGER NOT NULL, 
	continut TEXT NOT NULL, 
	creat_la DATETIME DEFAULT CURRENT_TIMESTAMP, 
	PRIMARY KEY (comentariu_id),
	FOREIGN KEY(activitate_id) REFERENCES activitati_club (activitate_id) ON DELETE CASCADE,
	FOREIGN KEY(user_id) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS club_threads;
CREATE TABLE club_threads (
	thread_id INTEGER NOT NULL AUTO_INCREMENT, 
	titlu VARCHAR(255) NOT NULL, 
	continut TEXT NOT NULL, 
	creat_de INTEGER NOT NULL, 
	creat_la DATETIME DEFAULT CURRENT_TIMESTAMP, 
	aprobat BOOL DEFAULT 0,
	PRIMARY KEY (thread_id),
	FOREIGN KEY(creat_de) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS thread_comments;
CREATE TABLE thread_comments (
	comentariu_id INTEGER NOT NULL AUTO_INCREMENT, 
	thread_id INTEGER NOT NULL, 
	user_id INTEGER NOT NULL, 
	continut TEXT NOT NULL, 
	likes INTEGER DEFAULT 0, 
	creat_la DATETIME DEFAULT CURRENT_TIMESTAMP, 
	PRIMARY KEY (comentariu_id),
	FOREIGN KEY(thread_id) REFERENCES club_threads (thread_id) ON DELETE CASCADE,
	FOREIGN KEY(user_id) REFERENCES users (user_id) ON DELETE CASCADE
);

DROP TABLE IF EXISTS thread_subcomments;
CREATE TABLE thread_subcomments (
	subcomentariu_id INTEGER NOT NULL AUTO_INCREMENT, 
	comentariu_id INTEGER NOT NULL, 
	user_id INTEGER NOT NULL, 
	continut TEXT NOT NULL, 
	likes INTEGER DEFAULT 0, 
	creat_la DATETIME DEFAULT CURRENT_TIMESTAMP, 
	PRIMARY KEY (subcomentariu_id),
	FOREIGN KEY(comentariu_id) REFERENCES thread_comments (comentariu_id) ON DELETE CASCADE,
	FOREIGN KEY(user_id) REFERENCES users (user_id) ON DELETE CASCADE
);

SET FOREIGN_KEY_CHECKS = 1;
